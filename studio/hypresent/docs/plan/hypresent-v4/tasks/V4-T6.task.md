```js
sizeCandidates = getElementGuidelines(el)
  .map((c) => { const r = c.getBoundingClientRect(); return { el: c, width: r.width, height: r.height }; })
  .filter((c) => c.width > 0 && c.height > 0)
  .filter((c) => {
    // F5 exclusion: drop a same-parent flex-grow sibling (its rendered size is the
    // flex solver's grow-share output, not an author-set size). Same-parent flex

    const cand = c.el;
    if (cand.parentElement === el.parentElement) {
      const pd = el.parentElement ? getComputedStyle(el.parentElement).display : "";
      if ((pd === "flex" || pd === "inline-flex") && parseFloat(getComputedStyle(cand).flexGrow) > 0) {

  });
```

## EDIT 3 — `showSizeHint(matchEl, value)` / `clearSizeHint()` (S3-9, scroll-offset-correct per G5)

ADD two functions. `showSizeHint` creates (once per gesture) a `hyp-size-hint` outline `<div>` positioned over `matchEl.getBoundingClientRect()` plus a `hyp-size-hint-badge` child showing `Math.round(value)+"px"`, appended to the interaction `wrapper`; `pointer-events:none`; dashed border. The wrapper is `position:absolute; top:0; left:0` spanning `scrollWidth×scrollHeight` (document-relative origin) but `getBoundingClientRect()` is viewport-relative — so the overlay position MUST add the scroll offset (mirror `positionMarker` in `comments.js`):

```js
let sizeHintEl = null;

function showSizeHint(matchEl, value) {
  const rect = matchEl.getBoundingClientRect();
  if (!sizeHintEl) {
    sizeHintEl = document.createElement("div");
    sizeHintEl.className = "hyp-size-hint";
    sizeHintEl.style.position = "absolute";
    sizeHintEl.style.pointerEvents = "none";
    sizeHintEl.style.border = "1px dashed #6366f1";
    sizeHintEl.style.boxSizing = "border-box";
    sizeHintEl.style.zIndex = "9";
    const badge = document.createElement("div");
    badge.className = "hyp-size-hint-badge";
    badge.style.position = "absolute";
    badge.style.fontSize = "11px";
    badge.style.background = "#6366f1";
    badge.style.color = "#fff";
    badge.style.padding = "1px 4px";
    badge.style.pointerEvents = "none";
    sizeHintEl.appendChild(badge);
    wrapper.appendChild(sizeHintEl);
  }
  // viewport rect → document-relative (wrapper origin is document-relative): add scrollX/scrollY
  sizeHintEl.style.left = (rect.left + window.scrollX) + "px";
  sizeHintEl.style.top = (rect.top + window.scrollY) + "px";
  sizeHintEl.style.width = rect.width + "px";
  sizeHintEl.style.height = rect.height + "px";
  sizeHintEl.firstChild.textContent = Math.round(value) + "px";
}

function clearSizeHint() {
  if (sizeHintEl && sizeHintEl.parentNode) sizeHintEl.parentNode.removeChild(sizeHintEl);
  sizeHintEl = null;
}
```

`wrapper` is the existing interaction wrapper element (`#hyp-interaction-wrapper`) — use the existing local reference; if it is reachable as `document.getElementById("hyp-interaction-wrapper")` only, use that. The `hyp-size-hint` namespace makes the overlay serializer-exempt (the existing `hyp-` namespace strip removes it). The position MUST use `window.scrollX/scrollY` per G5 (without it the hint floats to the wrong place on a scrolled deck — E-R11-3 asserts the overlap).
